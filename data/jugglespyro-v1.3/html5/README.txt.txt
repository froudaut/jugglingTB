Some Variables may be defined to modify the behavior of JuggleSpyro before calling the canevas:
These are, with its default values :

- JSpyroRightObjHandleSize = 4;
- JSpyroRightColorObjHandle = 'rgba(0,0,0,1)';
- JSpyroRightPropPathSize = 4;
- JSpyroRightHandPathSize = 4;
- JSpyroRightPoiStringSize = 2;
- JSpyroRightPoiRadius = 8;
- JSpyroRightObjStrokeSize = 3;

- JSpyroLeftObjHandleSize = 4;
- JSpyroLeftColorObjHandle = 'rgba(0,0,0,1)';
- JSpyroLeftPropPathSize = 4;
- JSpyroLeftHandPathSize = 4;
- JSpyroLeftPoiStringSize = 2;
- JSpyroLeftPoiRadius = 8;
- JSpyroLeftObjStrokeSize = 3;

- JSpyroGoPage = "two";        // Set it to "one" or "two"
- JSpyroColorBackground = 'rgba(153,255,102,1)';
- JSpyroScaling = 1;
- JSpyroSpeed = 80;
- JSpyroGIFQuality = 10;
- JSpyroGIFWorkers = 2;
- JSpyroGIFWidth = 400;
- JSpyroGIFHeight = 300;
- JSpyroGIFFrameDelay = 60;
- JSpyroGIFGrain = 4;
- JSpyroHelpURL = './help/help.html';
- JSpyroClockSpeedRatio = 1;
- JSpyroSetClockSpeedRatio = true;
- JSpyroCleaningMode = true;
- JSpyroRandomMode = false;
- JSpyroXMLSceneMode = true;
- JSpyroLoadXMLFileInput = "..."; // Set it to the default XML Animation Content wanted  
- JSpyroNbColorTransform = 1; // Enhanced this value to reduce the ColorTransform Filter usage on object after drawing
